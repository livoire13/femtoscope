 20def check_mode_qp(mat_func):
 21    """Decorator for not repeating the check on 'mode' parameter in each
 22    function definition."""
 23    @wraps(mat_func)
 24    def wrapper(*args, **kwargs):
 25        mode = kwargs.get('mode')
 26        if mode != 'qp':
 27            return
 28        return mat_func(*args, **kwargs)
 29
 30    return wrapper

 33class AbstractMaterials(ABC):
 34    """
 35    Template class for storing material functions of a certain type.
 36    """
 37
 38    def get_material(self, dim, coorsys=None, tag='int') -> Callable:
 39        """
 40        Retrieve material function according to input parameters.
 41
 42        Parameters
 43        ----------
 44        dim : int
 45            Dimension of the problem.
 46        coorsys : str
 47            Coordinate system. For valid strings, see `valid_coorsys`
 48        tag : str
 49            'int' for interior domain and 'ext' for exterior domain.
 50
 51        Returns
 52        -------
 53        func : Callable
 54            Material function matching input parameters.
 55
 56        """
 57
 58        _check_parameters(dim, coorsys, tag)  # perform checks
 59
 60        if dim == 1:  # Dimension 1
 61            if coorsys == 'polar':
 62                if tag == 'int':
 63                    func = self.mat1dpolint
 64                else:
 65                    func = self.mat1dpolext
 66            elif coorsys == 'cartesian':
 67                if tag == 'int':
 68                    func = self.mat3dint
 69                else:
 70                    func = self.mat1dcartext
 71            else:
 72                raise NotImplementedError
 73
 74        elif dim == 2:  # Dimension 2
 75            if coorsys == 'polar':
 76                if tag == 'int':
 77                    func = self.mat2dpolint
 78                else:
 79                    func = self.mat2dpolext
 80            elif coorsys == 'cylindrical':
 81                if tag == 'int':
 82                    func = self.mat2dcylint
 83                else:
 84                    func = self.mat2dcylext
 85            elif coorsys == 'cartesian':
 86                if tag == 'int':
 87                    func = self.mat3dint
 88                else:
 89                    func = self.mat2dcartext
 90            else:
 91                raise ValueError(f"Invalid coordinate system '{coorsys}'.")
 92
 93        else:  # Dimension 3
 94            if coorsys == 'cartesian':
 95                if tag == 'int':
 96                    func = self.mat3dint
 97                else:
 98                    func = self.mat3dext
 99            else:
100                raise NotImplementedError
101
102        return func

155class LaplacianMaterials(AbstractMaterials):
156    """
157    Shelf for storing material functions related to the laplacian term.
158    Whatever the dimension and/or the coordinate system, we consider the Laplace
159    operator in three dimensions.
160    """
161
162    # 1D material functions
163    @staticmethod
164    @check_mode_qp
165    def mat1dpolint(ts, coors, mode=None, **kwargs):
166        r = coors.squeeze()
167        val = r ** 2
168        return {'val': val.reshape(-1, 1, 1)}
169
170    @staticmethod
171    @check_mode_qp
172    def mat1dpolext(ts, coors, mode=None, Rc=None, **kwargs):
173        eta = coors.squeeze()
174        val = eta ** 4 / Rc ** 2 * (5 - 4 * eta / Rc)
175        return {'val': val.reshape(-1, 1, 1)}
176
177    @staticmethod
178    @check_mode_qp
179    def mat1dcartext(ts, coors, mode=None, Rc=None, **kwargs):
180        xi = coors.squeeze()
181        val = (xi / Rc) ** 4 * (3 - 2 * abs(xi) / Rc)
182        return {'val': val.reshape(-1, 1, 1)}
183
184    # 2D polar coordinates material functions
185    @staticmethod
186    @check_mode_qp
187    def mat2dpolint(ts, coors, mode=None, **kwargs):
188        val = np.zeros((coors.shape[0], 2, 2))
189        r, theta = coors[:, 0], coors[:, 1]
190        val[:, 0, 0] = sin(theta) * r ** 2
191        val[:, 1, 1] = sin(theta)
192        return {'val': val}
193
194    @staticmethod
195    @check_mode_qp
196    def mat2dpolext(ts, coors, mode=None, Rc=None, **kwargs):
197        val = np.zeros((coors.shape[0], 2, 2))
198        eta, theta = coors[:, 0], coors[:, 1]
199        val[:, 0, 0] = eta ** 4 / Rc ** 2 * (5 - 4 * eta / Rc) * sin(theta)
200        val[:, 1, 1] = eta ** 2 / Rc ** 2 * (5 - 4 * eta / Rc) * sin(theta)
201        return {'val': val}
202
203    # 2D cyclindrical coordinates material functions
204    @staticmethod
205    @check_mode_qp
206    def mat2dcylint(ts, coors, mode=None, **kwargs):
207        x = coors[:, 0]
208        val = abs(x).reshape(-1, 1, 1)
209        return {'val': val}
210
211    @staticmethod
212    @check_mode_qp
213    def mat2dcylext(ts, coors, mode=None, Rc=None, **kwargs):
214        xi, eta = coors[:, 0], coors[:, 1]
215        norm2 = xi ** 2 + eta ** 2
216        norm = sqrt(norm2)
217        val = norm2 ** 2 / Rc ** 4 * abs(xi) * (7 - 6 * norm / Rc)
218        return {'val': val.reshape(-1, 1, 1)}
219
220    @staticmethod
221    @check_mode_qp
222    def mat2dcartext(ts, coors, mode=None, Rc=None, **kwargs):
223        norm = np.linalg.norm(coors, axis=1)
224        val = (norm / Rc) ** 4 * (5 - 4 * norm / Rc)
225        return {'val': val.reshape(-1, 1, 1)}
226
227    # 3D cartesian coordinates material functions
228    mat3dint = None
229
230    @staticmethod
231    @check_mode_qp
232    def mat3dext(ts, coors, mode=None, Rc=None, **kwargs):
233        xi, eta, zeta = coors[:, 0], coors[:, 1], coors[:, 2]
234        norm2 = xi ** 2 + eta ** 2 + zeta ** 2
235        norm = sqrt(norm2)
236        val = norm2 ** 2 / Rc ** 4 * (7 - 6 * norm / Rc)
237        return {'val': val.reshape(-1, 1, 1)}
