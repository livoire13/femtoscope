 20def check_mode_qp(mat_func):
 21    """Decorator for not repeating the check on 'mode' parameter in each
 22    function definition."""
 23    @wraps(mat_func)

 42        Parameters

 58        _check_parameters(dim, coorsys, tag)  # perform checks

 61            if tag == 'int':
 62                func = self.mat1dint
 63            else:
 64                func = self.mat1dext
 65
 66        elif dim == 2:  # Dimension 2
 67            if coorsys == 'polar':
 68                if tag == 'int':
 69                    func = self.mat2dpolint
 70                else:
 71                    func = self.mat2dpolext
 72            elif coorsys == 'cylindrical':
 73                if tag == 'int':
 74                    func = self.mat2dcylint
 75                else:

 80                    func = self.mat3dint
 81                else:
 82                    raise NotImplementedError

 84            else:
 85                raise ValueError(f"Invalid coordinate system '{coorsys}'.")
 86
 87        else:  # Dimension 3

 89                if tag == 'int':
 90                    func = self.mat3dint
 91                else:

139class LaplacianMaterials(AbstractMaterials):
140    """
141    Shelf for storing material functions related to the laplacian term.
142    Whatever the dimension and/or the coordinate system, we consider the Laplace
143    operator in three dimensions.
144    """
145
146    # 1D material functions
147    @staticmethod
148    @check_mode_qp
149    def mat1dint(ts, coors, mode=None, **kwargs):
150        r = coors.squeeze()
151        val = r ** 2
152        return {'val': val.reshape(-1, 1, 1)}
153
154    @staticmethod
155    @check_mode_qp
156    def mat1dext(ts, coors, mode=None, Rc=None, **kwargs):
157        eta = coors.squeeze()
158        val = 20 * eta ** 3 / Rc ** 2 * (1 - eta / Rc)
159        return {'val': val.reshape(-1, 1, 1)}
160
161    # 2D polar coordinates material functions
162    @staticmethod
163    @check_mode_qp
164    def mat2dpolint(ts, coors, mode=None, **kwargs):
165        val = np.zeros((coors.shape[0], 2, 2))
166        r, theta = coors[:, 0], coors[:, 1]
167        val[:, 0, 0] = sin(theta) * r ** 2
168        val[:, 1, 1] = sin(theta)
169        return {'val': val}
170
171    @staticmethod
172    @check_mode_qp
173    def mat2dpolext(ts, coors, mode=None, Rc=None, **kwargs):
174        val = np.zeros((coors.shape[0], 2, 2))
175        eta, theta = coors[:, 0], coors[:, 1]
176        val[:, 0, 0] = eta ** 4 / Rc ** 2 * (5 - 4 * eta / Rc) * sin(theta)
177        val[:, 1, 1] = eta ** 2 / Rc ** 2 * (5 - 4 * eta / Rc) * sin(theta)
178        return {'val': val}
179
180    # 2D cyclindrical coordinates material functions
181    @staticmethod
182    @check_mode_qp
183    def mat2dcylint(ts, coors, mode=None, **kwargs):
184        x = coors[:, 0]
185        val = abs(x).reshape(-1, 1, 1)
186        return {'val': val}
187
188    @staticmethod
189    @check_mode_qp
190    def mat2dcylext(ts, coors, mode=None, Rc=None, **kwargs):
191        xi, eta = coors[:, 0], coors[:, 1]
192        norm2 = xi ** 2 + eta ** 2
193        norm = sqrt(norm2)
194        val = norm2 ** 2 / Rc ** 4 * abs(xi) * (7 - 6 * norm / Rc)
195        return {'val': val.reshape(-1, 1, 1)}
196
197    # 3D cartesian coordinates material functions
198    mat3dint = None
199
200    @staticmethod
201    @check_mode_qp
202    def mat3dext(ts, coors, mode=None, Rc=None, **kwargs):
203        xi, eta, zeta = coors[:, 0], coors[:, 1], coors[:, 2]
204        norm2 = xi ** 2 + eta ** 2 + zeta ** 2
205        norm = sqrt(norm2)
206        val = norm2 ** 2 / Rc ** 4 * (7 - 6 * norm / Rc)
207        return {'val': val.reshape(-1, 1, 1)}
