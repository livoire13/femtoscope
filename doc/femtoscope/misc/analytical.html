 339    def eval_mass(self, verbose=False):
 340        """Evaluation of the total mass of the source based on the integration
 341        of the denisty function."""
 342
 343        if self.coorsys == 'cartesian':
 344            def _integrand(x, y, z):
 345                return self.rho(x, y, z)*self.determinant(x, y, z)
 346
 347        elif self.coorsys == 'spherical':
 348            def _integrand(r, theta, phi):
 349                return self.rho(r, theta, phi)*self.determinant(r, theta, phi)
 350
 351        elif self.coorsys == 'spherical_mu':
 352            def _integrand(r, mu, phi):
 353                return self.rho(r, mu, phi)*self.determinant(r, mu, phi)
 354
 355        elif self.coorsys == 'cylindrical':
 356            def _integrand(z, r, phi):
 357                return self.rho(r, phi, z)*self.determinant(r, phi, z)
 358
 359        else:
 360            raise ValueError("Unknown coordinate system!")
 361
 362        mass, err  =  tplquad(_integrand, self.a, self.b, self.gfun,
 363                              self.hfun, self.qfun, self.rfun)
 364        if verbose:
 365            print("mass = {:.5e} ; err = {:.2e}".format(mass, err))
 366        return mass, err

 368    def eval_potential(self, coor1, coor2, coor3, verbose=False):
 369        """Evaluation of the gravitational potential at coordinates
 370        (`coor1`, `coor2`, `coor3`) in a pre-determined coordinate system.
 371        Acts as a wrapper for `_eval_pot_point`."""
 372
 373        if isinstance(coor1, (list, np.ndarray, tuple)):
 374            pots, errs = [], []
 375            for kk in range(len(coor1)):
 376                res = self._eval_pot_point(coor1[kk], coor2[kk], coor3[kk])
 377                pots.append(-res[0])
 378                errs.append(+res[1])
 379                if verbose:
 380                    PotentialByIntegration.print_coors(
 381                        self.coorsys, coor1[kk], coor2[kk], coor3[kk])
 382                    print("pot = {:.5e} ; err = {:.2e}\n".format(pots[-1], errs[-1]))
 383            return np.array(pots), np.array(errs)
 384
 385        elif np.isscalar(coor1):
 386            res = self._eval_pot_point(coor1, coor2, coor3)
 387            pot, err = -res[0], res[1]
 388            if verbose:
 389                PotentialByIntegration.print_coors(self.coorsys, coor1, coor2, coor3)
 390                print("pot = {:.5e} ; err = {:.2e}".format(pot, err))
 391            return pot, err
 392
 393        else:
 394            raise ValueError("coordinates must be specified as scalars or lists")

 396    def _eval_pot_point(self, coor1, coor2, coor3):
 397        """Evaluation of potential at point (coor1, coor2, coor3)."""
 398
 399        if self.coorsys == 'cartesian':
 400            def _integrand(x, y, z):
 401                func = partial(self.integrand, xeval=coor1,
 402                               yeval=coor2, zeval=coor3)
 403                return func(x, y, z)
 404
 405        elif self.coorsys == 'spherical':
 406            def _integrand(r, theta, phi):
 407                func = partial(self.integrand, reval=coor1, thetaeval=coor2,
 408                               phieval=coor3)
 409                return func(r, theta, phi)
 410
 411        elif self.coorsys == 'spherical_mu':
 412            def _integrand(r, mu, phi):
 413                func = partial(self.integrand, reval=coor1, mueval=coor2,
 414                               phieval=coor3)
 415                return func(r, mu, phi)
 416
 417
 418        elif self.coorsys == 'cylindrical':
 419            def _integrand(z, r, phi):
 420                func = partial(self.integrand, reval=coor1, phieval=coor2,
 421                               zeval=coor3)
 422                return func(r, phi, z)
 423
 424        else:
 425            raise ValueError(
 426                "{} is not a known coordinate system!".format(self.coorsys))
 427
 428        return tplquad(_integrand, self.a, self.b, self.gfun, self.hfun,
 429                       self.qfun, self.rfun)

 445def potential_sphere(r, R, G, M=None, rho=None):
 446    """
 447    Gravitational potential created by a perfect homogeneous solid
 448    sphere. Note that `M` and `rho` cannot be simultaneously None.
 449
 450    Parameters
 451    ----------
 452    r : float or ndarray
 453        Radial distance from the mass center.
 454    R : float
 455        Radius of the sperical body.
 456    G : float
 457        Gravitational constant.
 458    M : float, optional
 459        Mass of the body. The default is None.
 460    rho : float
 461        Density of the body. The default is None.
 462
 463    Returns
 464    -------
 465    ndarray
 466        Gravitational potential at radii `r` from the center of the spherical
 467        body.
 468
 469    See Also
 470    --------
 471    potential_ellipsoid : generalization to an oblate ellipsoid.
 472
 473    """
 474    r = np.array(r).reshape(-1)
 475    assert M is not None or rho is not None
 476    if rho is not None:
 477        M = (rho*4*pi*R**3)/3
 478    with np.errstate(divide='ignore'):
 479        pot = np.where(r<R, 0.5*G*M*((r/R)**2-3)/R, -G*M/r)
 480    if len(pot) == 1: pot = pot[0]
 481    return pot

 514def _potential_sphere_eta(eta, R, G, Rcut, M=None, rho=None):
 515    r"""
 516    Gravitational potential created by a perfect homogeneous solid
 517    sphere in the `eta` coordinate. Note that `M` and `rho` cannot be
 518    simultaneously None.
 519
 520    $$ \eta = \dfrac{Rcut}{r} $$
 521
 522    Parameters
 523    ----------
 524    eta : float or ndarray

 526    R : float
 527        Radius of the sperical body.
 528    G : float
 529        Gravitational constant.
 530    Rcut : float
 531        Truncation radius (appearing in the definition of eta).
 532    M : float, optional
 533        Mass of the body. The default is None.
 534    rho : float
 535        Density of the body. The default is None.
 536
 537    Returns
 538    -------
 539    ndarray
 540        Gravitational potential at coordinate `eta` from the center of the
 541        spherical body.
 542    """
 543    eta = np.array(eta).reshape(-1)
 544    for x in eta:
 545        if x > Rcut*1.01:
 546            import warnings
 547            warnings.warn("""eta is usually set below Rcut.
 548                          Consider using `potential_sphere` in this case""")

 550    if rho is not None:
 551        M = (rho*4*pi*R**3)/3
 552    return -G*M*eta/Rcut**2

 602    coors_cart : ndarray
 603        Cartesian coordinates (numpy array (:, 3)-shapped).
 604    sa : float
 605        Spheroid semi-major-axis.
 606    G : float
 607        Gravitational constant.
 608    ecc : float, optional
 609        Spheroid eccentricity. The default is None.
 610    sc : float, optional
 611        Spheroid semi-minor-axis. The default is None.
 612    M : float, optional
 613        Mass of the body. The default is None.
 614    rho : float, optional
 615        Density of the body. The default is None.
 616
 617    Returns
 618    -------
 619    ndarray
 620        Gravitational potential at `coors` locations.
 621
 622    See Also
 623    --------
 624    potential_sphere : Particular case of the perfect homogeneous sphere.
 625
 626    References
 627    ----------
 628    This implementation rely on [1]_ for the potential inside a homogeneous
 629    ellipsoid of revolution and [2]_ for the potential inside & outside.
 630
 631    .. [1] DOI:10.1093/oso/9780198786399.003.0015.
 632
 633    .. [2] HVOÅ½DARA, M., & KOHÃT, I. (2011). "Gravity field due to a
 634           homogeneous oblate spheroid: Simple solution form and numerical
 635           calculations". Contributions to Geophysics and Geodesy, 41(4),
 636           307-327. https://doi.org/10.2478/v10126-011-0013-0.
 637
 638    """
 639    coors_cart = np.array(coors_cart)
 640    if len(coors_cart.shape)==1:
 641        coors_cart = coors_cart[:, np.newaxis].T
 642    X, Y, Z = coors_cart[:,0], coors_cart[:,1], coors_cart[:,2]
 643
 644    # Assertions
 645    assert ecc is not None or sc is not None
 646    assert M is not None or rho is not None
 647
 648    # Some conversions
 649    if ecc is None:
 650        ecc = sqrt(1-(sc/sa)**2)
 651    elif sc is None:
 652        sc = sa*sqrt(1-ecc**2)
 653    if M is None:
 654        M = 4*rho*pi*sc*sa**2/3
 655    elif rho is None:
 656        rho = 3*M/(4*pi*sc*sa**2)
 657    boolin = (X**2+Y**2)/sa**2 + Z**2/sc**2 < 1.0
 658
 659    # Some function definitions
 660    def P2(t):
 661        return 0.5 * (3*t**2 - 1)
 662    def P2i(t):
 663        return -0.5 * (3*t**2 + 1)
 664    def q2(t):
 665        return 0.5 * ((3*t**2 + 1)*arctan(1/t) - 3*t)
 666    def q2prime(t):
 667        return -(2+3*t**2)/(1+t**2) + 3*t*arctan(1/t)
 668    def potin(w0, chalpha, cosbeta, E0, E2, shalpha):
 669        return -w0*(chalpha**2*(P2(cosbeta)-1)+E0+E2*P2i(shalpha)*P2(cosbeta))
 670    def potout(shalpha, cosbeta, f):
 671        return -(G*M*(arctan(1/shalpha) + q2(shalpha) * P2(cosbeta))/f)
 672
 673    # Some coefficients
 674    f = sqrt(sa**2 - sc**2) # f = sa*ecc
 675    E0 = (sa/f)**2 * (1 + 2*(sc/f)*arctan(f/sc))
 676    ch02 = (sa/f)**2
 677    sh0 = sc/f
 678    w0 = 2*pi*G*rho*f**2 / 3
 679    E2 = -ch02 * (ch02*q2prime(sh0) - 2*sh0*q2(sh0))
 680    r = sqrt(X**2 + Y**2)
 681    chalpha = 0.5 * (sqrt((r-f)**2 + Z**2) + sqrt((r+f)**2 + Z**2)) / f
 682    shalpha = sqrt(chalpha**2 - 1)
 683    cosbeta = Z / (f*shalpha)
 684
 685    # Formula from Maclaurin
 686    I = 2*arcsin(ecc)/ecc
 687    A1 = (arcsin(ecc)-ecc*sqrt(1-ecc**2))/ecc**3
 688    A3 = 2*(ecc-sqrt(1-ecc**2)*arcsin(ecc))/(sqrt(1-ecc**2)*ecc**3)
 689    potmac = -pi*G*rho*sqrt(1-ecc**2)*(sa**2*I-(X**2+Y**2)*A1-Z**2*A3)
 690
 691    # return np.where(boolin, potmac, potout(shalpha, cosbeta, f))
 692    return np.where(boolin, potmac, potout(shalpha, cosbeta, f))

 695def chameleon_radial(r, R_A, rho_in, rho_vac, alpha, npot, plot=False,
 696                     verbose=False):
 697    """
 698    Approximate analytical solution of the chameleon field within and around a
 699    perfect homogeneous solid sphere.
 700
 701    Parameters
 702    ----------
 703    r : float or ndarray
 704        Radial distance from the mass center.
 705    R_A : float
 706        Radius of the sperical body.
 707    rho_in : float
 708        Density inside the spherical body.
 709    rho_vac : float
 710        Vacuum density.
 711    alpha : float
 712        Physical parameter weighting the laplacian operator of the Klein-Gordon
 713        equation (dimensionless).
 714    npot : int
 715        Exponent (parameter of the chameleon model).
 716    plot : bool
 717        Whether or not to plot the resulting profile. The default is False.
 718    verbose : bool
 719        Display user's information. The default is False.
 720
 721    Returns
 722    -------
 723    phi : ndarray
 724        Chameleon scalar field at distance `r` from the body center.
 725
 726    Notes
 727    -----
 728    The solution returned by this function is only an approximate solution and
 729    by no means should be used to compute the *error* of a FEM result for
 730    instance.
 731
 732    References
 733    ----------
 734    See [1]_ or [2]_ for the analytical derivation of the solution of the
 735    Klein-Gordon equation.
 736
 737    .. [1] "Chameleon cosmology", Justin Khoury and Amanda Weltman,
 738           Phys. Rev. D 69, 044026 â Published 27 February 2004.
 739
 740    .. [2] "Testing gravity in space : towards a realistic treatment of
 741           chameleon gravity in the MICROSCOPE mission"", Martin Pernot-BorrÃ s,
 742           PhD thesis manuscript, November 2020.
 743
 744    """
 745    r = np.array(r).reshape(-1)
 746    phi = np.zeros_like(r, dtype=np.float64)
 747    m_vac = np.sqrt((npot+1)/alpha * rho_vac**((npot+2)/(npot+1)))
 748    phi_in = rho_in**(-1/(npot+1))
 749    phi_vac = rho_vac**(-1/(npot+1))
 750    poly = np.array([m_vac/(3*alpha*(1+m_vac*R_A)), -1/(2*alpha), 0,
 751                     R_A**2/(6*alpha)*(2/(1+m_vac*R_A)+1)
 752                     + (phi_in-phi_vac)/rho_in])
 753    roots = np.roots(poly)
 754    roots = np.real(roots[np.where(np.isreal(roots))[0]])
 755    thin_shell = np.where((roots>=0) & (roots<=R_A))[0].size != 0
 756
 757    # Thin-shell regime
 758    if thin_shell:
 759        if verbose: print("thin-shell")
 760        R_TS = roots[np.where((roots>=0) & (roots<=R_A))[0]]
 761        # Keep the solution that makes K>0
 762        R_TS = np.double(R_TS[np.where(R_TS<=R_A)].squeeze())
 763        K = rho_in * (R_A**3 - R_TS**3) / (3*alpha*(1+R_A*m_vac))
 764        r2 = r[np.where((r>=R_TS) & (r<=R_A))[0]]
 765        r3 = r[np.where(r>R_A)[0]]
 766        phi[np.where(r<R_TS)[0]] = phi_in
 767        phi[np.where((r>=R_TS) & (r<=R_A))[0]] = phi_in + rho_in/(3*alpha) * \
 768            (r2**2/2 + R_TS**3/r2 - 3*R_TS**2/2)
 769        phi[np.where(r>R_A)[0]] = phi_vac - K/r3 * np.exp(-m_vac*(r3-R_A))
 770
 771    # Thick-shell regime
 772    else:
 773        if verbose: print("thick-shell")
 774        K = rho_in/(3*alpha) * R_A**3 / (1 + m_vac*R_A)
 775        phi_min = phi_vac - K/R_A - rho_in*R_A**2/(6*alpha)
 776        r1 = r[np.where(r<R_A)[0]]
 777        r2 = r[np.where(r>=R_A)[0]]
 778        phi[np.where(r<R_A)[0]] = phi_min + rho_in/(6*alpha) * r1**2
 779        phi[np.where(r>=R_A)[0]] = phi_vac - K/r2 * np.exp(-m_vac*(r2-R_A))
 780
 781    if plot:
 782        from matplotlib import pyplot as plt
 783        plt.figure()
 784        plt.plot(r, phi, color='black')
 785        plt.show()
 786
 787    return phi

 789def thin_shell(r, R_A, rho_in, rho_vac, alpha, npot):
 790    """Compute the Thin-Shell thickness [see `chameleon_radial` function for
 791    full documentation]."""
 792    r = np.array(r).reshape(-1)
 793    phi = np.zeros_like(r, dtype=np.float64)
 794    m_vac = np.sqrt((npot+1)/alpha * rho_vac**((npot+2)/(npot+1)))
 795    phi_in = rho_in**(-1/(npot+1))
 796    phi_vac = rho_vac**(-1/(npot+1))
 797    poly = np.array([m_vac/(3*alpha*(1+m_vac*R_A)), -1/(2*alpha), 0,
 798                     R_A**2/(6*alpha)*(2/(1+m_vac*R_A)+1)
 799                     + (phi_in-phi_vac)/rho_in])
 800    roots = np.roots(poly)
 801    roots = np.real(roots[np.where(np.isreal(roots))[0]])
 802    thin_shell = np.where((roots>=0) & (roots<=R_A))[0].size != 0
 803
 804    # Thin-shell regime
 805    if thin_shell:
 806        R_TS = roots[np.where((roots>=0) & (roots<=R_A))[0]]
 807        R_TS = np.double(R_TS[np.where(R_TS<=R_A)].squeeze())
 808        return R_A - R_TS
 809
 810    # Thick-shell regime
 811    else:
 812        return R_A

 979def _iso_alpha_bounds(iso, Lambda_bounds, beta_bounds, npot, L_0, rho_0):
 980    r""" Utility function for plotting curves of iso-alpha on alpha-maps. The
 981    equation reads
 982            $$ iso = A + \frac{n+4}{n+1} y - \frac{n+2}{n+1} x $$
 983    Iso-alpha curves are straight lines in the $(\log \beta , \log \Lambda)$
 984    plane."""
 985    from femtoscope.misc import constants
 986    from femtoscope.misc.constants import H_BAR, C_LIGHT, EV
 987    M_pl = constants.M_PL
 988    beta_min = np.log10(beta_bounds[0])
 989    beta_max = np.log10(beta_bounds[1])
 990    Lambda_min = np.log10(Lambda_bounds[0])
 991    Lambda_max = np.log10(Lambda_bounds[1])
 992
 993    # Compute limits
 994    A = np.log10(M_pl/(rho_0*L_0**2) * EV/(H_BAR*C_LIGHT))
 995    A += 1/(npot+1)*np.log10(npot*M_pl/rho_0 * (EV/(H_BAR*C_LIGHT))**3)
 996    xmin = (npot+1)/(npot+2) * (A - iso + (npot+4)/(npot+1)*Lambda_min)
 997    xmax = (npot+1)/(npot+2) * (A - iso + (npot+4)/(npot+1)*Lambda_max)
 998    ymin = (npot+1)/(npot+4) * (iso - A + (npot+2)/(npot+1)*beta_min)
 999    ymax = (npot+1)/(npot+4) * (iso - A + (npot+2)/(npot+1)*beta_max)
1000
1001    # crop to get the correct limits
1002    xmin = beta_min if xmin<beta_min else xmin
1003    xmax = beta_max if xmax>beta_max else xmax
1004    ymin = Lambda_min if ymin<Lambda_min else ymin
1005    ymax = Lambda_max if ymax>Lambda_max else ymax
1006    return ([xmin, xmax], [ymin, ymax])

 21    """

 23    integration (hence we refer to this method as 'semi-analytical').
 24
 25    Attributes
 26    ----------
 27    coorsys : str
 28        The set of coordinates to be used ('cartesian' or 'spherical' or

 32    denom : function

 34    integrand : function
 35        Function to be integrated with `scipy.integrate.tplquad`, which gives
 36        the gravitational potential value.
 37    rho : function
 38        Density function (arbitrary function with compact support).
 39    a, b, gfun, hfun, qfun, rfun : parameters (see scipy doc in the link below)
 40        https://docs.scipy.org/doc/scipy/reference/generated/scipy.integrate.tplquad.html
 41
 42    Methods
 43    -------
 44    eval_mass :
 45        Computes the total mass based on the mass distribution provided
 46        by the user.
 47    eval_potential :
 48        Computes the gravitational potential at the desired location(s).
 49
 50
 51    """
 52    def __init__(self, coorsys, preset=None, rho=1.0, a=None, b=None,
 53                 gfun=None, hfun=None, qfun=None, rfun=None, **kwargs):
 54        """
 55        Construct a `PotentialByIntegration` instance. If preset is set to None,
 56        all the subsequent keyword arguments must be specified.
 57
 58        Parameters
 59        ----------
 60        See above class docstring.
 61
 62        """
 63
 64        self.coorsys = coorsys
 65
 66        if coorsys == 'cartesian':
 67            self.determinant = lambda x, y, z : 1.0
 68            def denom(x1, x2, y1, y2, z1, z2):
 69                return sqrt((x1-x2)**2 + (y1-y2)**2 + (z1-z2)**2)
 70            def integrand(xp, yp, zp, xeval=None, yeval=None, zeval=None):
 71                return rho(xp, yp, zp)*self.determinant(xp, yp, zp) \
 72                    / denom(xeval, xp, yeval, yp, zeval, zp)

 76            def denom(r1, r2, theta1, theta2, phi1, phi2):

 78                    sin(theta1)*sin(theta2)*cos(phi1-phi2) \
 79                        + cos(theta1)*cos(theta2)))
 80            def integrand(rp, thetap, phip, reval=None, thetaeval=None,
 81                          phieval=None):
 82                return rho(rp, thetap, phip)*self.determinant(rp, thetap, phip) \
 83                    / denom(reval, rp, thetaeval, thetap, phieval, phip)
 84
 85        elif coorsys == 'spherical_mu':
 86            self.determinant = lambda r, mu, phi : r**2
 87            def denom(r1, r2, mu1, mu2, phi1, phi2):
 88                return sqrt(r1**2 + r2**2 - 2*r1*r2*(
 89                    sqrt((1-mu1**2)*(1-mu2**2))*cos(phi1-phi2) + mu1*mu2))
 90            def integrand(rp, mup, phip, reval=None, mueval=None, phieval=None):
 91                return rho(rp, mup, phip)*self.determinant(rp, mup, phip) \
 92                    / denom(reval, rp, mueval, mup, phieval, phip)
 93

 96            def denom(r1, r2, phi1, phi2, z1, z2):
 97                return sqrt(r1**2 + r2**2 - 2*r1*r2*cos(phi1-phi2) \
 98                            + (z1-z2)**2)
 99            def integrand(rp, phip, zp, reval=None, phieval=None, zeval=None):
100                return rho(rp, phip, zp)*self.determinant(rp, phip, zp) \
101                    / denom(reval, rp, phieval, phip, zeval, zp)
102
103        else:
104            raise ValueError(
105                "{} is not a known coordinate system!".format(coorsys))
106
107        self.denom = denom
108        self.integrand = integrand
109

112            rho_val = rho
113

Attributes
  • coorsys (str): The set of coordinates to be used ('cartesian' or 'spherical' or 'cylindrical').
  • determinant (function): Pre-factor of the volume element (coordinate-system-dependent).

  • rho (function): Density function (arbitrary function with compact support).
