 192    def solve(self, use_reduced_mtx_vec=True, use_buffer=False, **kwargs):
 193        """
 194        Solve the linear problem.
 195
 196        Parameters
 197        ----------
 198        use_reduced_mtx_vec : bool, optional
 199            Use reduced matrix and rhs vector for the linear system solving
 200            stage. The default is True.
 201        use_buffer : bool, optional
 202            Store the solution of the linear system in a buffer at the end of
 203            the linear system solving stage. This option is enabled when solving
 204            nonlinear problems with Newton's iterations.
 205            The default is False.
 206
 207        Other Parameters
 208        ----------------
 209        verbose : bool
 210            Display user information. The default is False.
 211
 212        """
 213
 214        verbose = kwargs.get('verbose', False)
 215
 216        if self.wf_ext is None:  # Single weak form case
 217            self._solve_single_wf(use_reduced_mtx_vec, use_buffer)
 218
 219        else:  # Two weak forms case
 220
 221            if self.connection_method == 'ifem':
 222                if not use_reduced_mtx_vec:
 223                    warnings.warn(f"Cannot use full mtx/vec with method "
 224                                  f"'{self.connection_method}'")
 225                self._solve_two_wf_ifem(
 226                    use_buffer, self.region_key_int, self.region_key_ext)
 227
 228            elif self.connection_method == 'a-ifem':
 229                self._solve_two_wf_a_ifem(
 230                    use_buffer, self.region_key_int, self.region_key_ext)
 231
 232            else:
 233                raise NotImplementedError(
 234                    f"connection method '{self.connection_method}' "
 235                    f"is not implemented!")

 266    def save_results_to_vtk(self, name: Union[None, str] = None):
 267        """
 268        Save the FEM results and set the attributes `result_int_name` and
 269        `result_ext_name` (when relevant) using `create_structured_vtk`.
 270        When `sol_ext` is not None, two separate files are created.
 271
 272        Parameters
 273        ----------
 274        name : Union[None, str], optional
 275            Name of the .vtk result file. If `sol_ext` is not None, the suffix
 276            '_int' / '_ext' is appended to the provided name in order to
 277            distinguish between the two created files. The default is None, in
 278            which case the current date-time is used.
 279
 280        """
 281
 282        if self.wf_int.dim == 1:  # handling of the one-dimensional case
 283            print("Results of 1D problems cannot be saved to VTK.")
 284            return
 285        aux_name = make_absolute_vtk_name(name)
 286        aux_name = str(Path(aux_name).with_suffix(''))
 287        for suffix in ('_int', '_ext'):
 288
 289            if self.wf_ext is None and suffix == '_ext':
 290                continue  # no exterior solution
 291
 292            # Prepare data for VTK creation
 293            if self.wf_ext is not None:
 294                path_name = aux_name + suffix
 295            else:
 296                path_name = aux_name
 297            wf = getattr(self, 'wf' + suffix)
 298            coors = wf.field.coors
 299            cells = create_connectivity_table(coors)
 300            vars_dict = {'sol' + suffix: getattr(self, 'sol' + suffix)}
 301
 302            # Residual vector & parts (when relevant!)
 303            if (suffix == '_int') and (not self.is_linear):
 304                monitor = self.nonlinear_monitor
 305                if monitor.residual_vetor is not None:
 306                    vars_dict['residual'] = monitor.residual_vetor
 307                if 'ResidualVectorParts' in monitor.criteria:
 308                    res_parts = monitor.criteria['ResidualVectorParts'].value
 309                    vars_dict['res_mtx'] = res_parts['mtx_term']
 310                    vars_dict['res_rhs_cst'] = res_parts['rhs_cst_term']
 311                    vars_dict['res_rhs_mod'] = res_parts['rhs_mod_term']
 312
 313            # VTK creation
 314            create_structured_vtk(coors, vars_dict, cells, path_name=path_name)
 315            setattr(self, 'result{}_name'.format(suffix), path_name + '.vtk')

 334    def _solve_single_wf(self, use_reduced_mtx_vec: bool, use_buffer: bool):
 335        """Solve the linear problem consisting of a single weak form."""
 336
 337        # Get matrix and rhs vector
 338        if use_reduced_mtx_vec:
 339            if self.wf.mtx_vec.mtx_bc_reduced is None:
 340                self.wf.set_mtx_vec_bc_reduced()
 341            mtx = self.wf.mtx_vec.mtx_bc_reduced
 342            rhs = self.wf.mtx_vec.vec_bc_reduced
 343        else:
 344            if self.wf.mtx_vec.get_mtx_bc_full(self.wf.pb_cst) is None:
 345                self.wf.set_mtx_vec_bc_full()
 346            mtx = self.wf.mtx_vec.get_mtx_bc_full(self.wf.pb_cst)
 347            rhs = self.wf.mtx_vec.get_vec_bc_full(self.wf.pb_cst)
 348
 349        # Solve linear system
 350        vec_x = self._solve_linear_system(mtx, rhs)
 351
 352        # Reconstruct full solution satisfying E(P)BCs
 353        if use_reduced_mtx_vec:
 354            vec_x = self.wf.make_full_vec(vec_x)
 355
 356        # Store solution
 357        if use_buffer:
 358            self.buffer_int[:] = vec_x
 359        else:
 360            self.sol_int[:] = vec_x

 362    def _solve_two_wf_ifem(self, use_buffer: bool,
 363                           region_key_int: tuple, region_key_ext: tuple):
 364        """Solve the linear problem consisting of two weak forms using the
 365        so-called 'inverted finite element method'."""
 366
 367        # Create new weak form with LCBCs
 368        wf_int = self.wf_int
 369        wf_ext = self.wf_ext
 370        wf = WeakForm.from_two_weak_forms(
 371            wf_int, wf_ext, region_key_int, region_key_ext)
 372
 373        # Get matrix and rhs vector
 374        if wf.mtx_vec.mtx_bc_reduced is None:
 375            wf.set_mtx_vec_bc_reduced()
 376        mtx = wf.mtx_vec.mtx_bc_reduced
 377        rhs = wf.mtx_vec.vec_bc_reduced
 378
 379        # Solve linear system
 380        vec_x = self._solve_linear_system(mtx, rhs)
 381
 382        # Reconstruct interior and exterior solutions
 383        variables_cst = wf.pb_cst.get_variables()
 384        variables_cst.set_state(vec_x, reduced=True, force=True)
 385        sol_int = variables_cst.get(wf_int.get_unknown_name('cst'))()
 386        sol_ext = variables_cst.get(wf_ext.get_unknown_name('cst'))()
 387
 388        # Store solution
 389        if use_buffer:
 390            self.buffer_int = sol_int
 391            self.buffer_ext = sol_ext
 392        else:
 393            self.sol_int[:] = sol_int
 394            self.sol_ext[:] = sol_ext

 402    def _solve_linear_system(self, mtx, rhs):
 403        status = IndexedStruct()
 404        eps_a = self.ls_kwargs.get('eps_a', LinearSolver.default_eps_a)
 405        eps_r = self.ls_kwargs.get('eps_r', LinearSolver.default_eps_r)
 406        max_iter_ls = self.ls_kwargs.get('i_max',
 407                                         LinearSolver.default_max_iter_ls)
 408        vec_x = self.ls(rhs, x0=None, eps_a=eps_a, eps_r=eps_r,
 409                        i_max=max_iter_ls, mtx=mtx, status=status)
 410        return vec_x

 444    @staticmethod
 445    def _check_wf_dict(wf_dict: dict):
 446        """
 447        Perform several sanity checks on the provided weak forms at solver
 448        instantiation.
 449
 450        Parameters
 451        ----------
 452        wf_dict : dict
 453            Dictionary containing `WeakForm` instances passed to the current
 454            class constructor.
 455        """
 456
 457        # Check types
 458        for wf in wf_dict.values():
 459            if not isinstance(wf, WeakForm):
 460                raise TypeError(f"'wf_dict' should only contain 'WeakForm'"
 461                                f"instances, but '{wf}' is of type {type(wf)}")
 462
 463        if 'wf_ext' not in wf_dict:
 464            for wf in wf_dict.values():
 465                if wf._is_exterior:
 466                    raise ValueError("weak forms should all be defined on the"
 467                                     "interior domain!")
 468                wf._is_exterior = False
 469
 470        else:
 471            wf_dict['wf_int']._is_exterior = False
 472            wf_dict['wf_ext']._is_exterior = True
 473            for key, wf in wf_dict.items():
 474                if key in ['wf_int', 'wf_ext']:
 475                    continue
 476                if wf._is_exterior is None:
 477                    raise ValueError(f"User must specify whether weak form"
 478                                     f"'{wf.name}' is interior or exterior.")
 479

 481    def _check_ls_class(ls_class: str):
 482        """Check the `ls_class` parameter at solver instantiation."""
 483        ls_class_dict = LinearSolver.ls_class_dict
 484        if ls_class not in ls_class_dict:
 485            raise KeyError(f"'ls_class' must be in {ls_class_dict}")

 487    @staticmethod

 673    def solve(self, verbose=True, **kwargs):
 674        """
 675        Solve the nonlinear problem.
 676        Override `solve` method from parent class `LinearSolver`.
 677
 678        Parameters
 679        ----------
 680        verbose : bool, optional
 681            Get information about the solving process. The default is True.
 682
 683        Other Parameters
 684        ----------------
 685        first_call : bool, optional
 686            Whether this method is called for the first time or it resumes.
 687            The default is False.
 688        pause_iter_num : int, optional
 689            If specified, pause the iterations at the corresponding iteration
 690            number. The default is -1.
 691        connection_method : str
 692            Selector of the method to be used for linking `wf_int` and `wf_ext`.
 693            The default is 'ifem'.
 694        region_key_int : tuple
 695            Key of the connecting region in the interior domain. Mandatory
 696            parameter for two-weak-form problems.
 697        region_key_ext : tuple
 698            Key of the connecting region in the exterior domain. Mandatory
 699            parameter for two-weak-form problems.
 700
 701        """
 702
 703        # Retrieve keyword arguments
 704        first_call = kwargs.get('first_call', True)
 705        pause_iter_num = kwargs.get('pause_iter_num', -1)
 706
 707        # Set up before entering main loop
 708        monitor = self.nonlinear_monitor
 709        if first_call:
 710            self._check_solver_monitor_link()
 711            self.update_nonlinear_materials('all')
 712            monitor.current_iter_num = 0
 713            monitor.evaluate_all_criteria()
 714        if verbose:
 715            monitor.display_criteria_info('look', 'active', 'threshold')
 716
 717        # Main 'while' loop
 718        while not monitor.stop:
 719            monitor.current_iter_num += 1
 720
 721            # Solve the linearized problem, the solution is put in the buffer
 722            LinearSolver.solve(
 723                self, use_reduced_mtx_vec=True, use_buffer=True, **kwargs)
 724            self.bound_solution(buffer=True)
 725
 726            # Relaxation step
 727            self.perform_relaxation_step()
 728            self.bound_solution(buffer=False)
 729
 730            # Update matrices and vectors with new solution
 731            self.update_nonlinear_mtx_vec('all')
 732
 733            # Monitor the current iteration
 734            monitor.evaluate_all_criteria()
 735            monitor.update_status()
 736            if verbose:
 737                monitor.display_criteria_info('value', 'threshold')
 738
 739            # Update 'old' solution
 740            self.old_int[:] = self.sol_int
 741            if self.sol_ext is not None:
 742                self.old_ext[:] = self.sol_ext
 743
 744            # Pause?
 745            if monitor.current_iter_num == pause_iter_num:
 746                self._write_pause_message(pause_iter_num)
 747                return
 748
 749        monitor.write_termination_status()
 750        monitor.display_iterations_report()

 752    def resume(self, force=False, new_maximum_iter_num=None,
 753               extra_iter_num=None, **kwargs):
 754        """
 755        Resume the iterations by calling `solve` method with `first_call=False`.
 756
 757        Parameters
 758        ----------
 759        force : bool, optional
 760            If True, set all stopping criteria as inactive except the maximum
 761            number of iterations 'MaximumIterations' which remains active.
 762        new_maximum_iter_num : int, optional
 763            The new threshold for criterion 'MaximumIterations'.
 764            The default is None.
 765        extra_iter_num : int, optional
 766            Number of additional iterations to undertake. The default is None.
 767        kwargs : dict
 768            Keyword arguments passed to the 'solve' method.
 769        """
 770
 771        # Retrieve keyword arguments and handle exception
 772        verbose = kwargs.get('verbose', True)
 773        if (new_maximum_iter_num is not None) and (extra_iter_num is not None):
 774            raise Warning("It is not possible to set 'new_maximum_iter_num' "
 775                          "and 'extra_iter_num' simultaneously, aborting.")
 776            return
 777
 778        # Current data
 779        monitor = self.nonlinear_monitor
 780        old_maximum_iter_num = monitor.maximum_iter_num
 781
 782        # Set a new maximum number of iterations
 783        if new_maximum_iter_num is not None:
 784            if new_maximum_iter_num <= old_maximum_iter_num:
 785                if verbose:
 786                    print("No additional iterations to perform!")
 787                return
 788            monitor.set_maximum_iter_num(new_maximum_iter_num)
 789        if extra_iter_num is not None:
 790            monitor.set_maximum_iter_num(old_maximum_iter_num + extra_iter_num)
 791
 792        if force:  # force the undertaking of iterations by disabling criteria
 793            monitor.disable_all_criteria()
 794            monitor.stop = False
 795
 796        self.solve(first_call=False, **kwargs)

 798    def bound_solution(self, buffer=False):
 799        """
 800        Bound the solution between `sol_bounds[0]` and `sol_bounds[1]` (when
 801        bounds have been provided to the solver). The `buffer` toggle (bool)
 802        indicates whether the correction is to be made on the buffer or on the
 803        current solution. Lower and/or upper bounds can be None, in which case
 804        the truncation is not performed.
 805        """
 806
 807        if self.sol_bounds is None:  # No bounds provided
 808            return
 809
 810        sol_min = self.sol_bounds[0]
 811        sol_max = self.sol_bounds[1]
 812
 813        if buffer:  # Use the buffer site
 814            sol_list = [self.buffer_int, self.buffer_ext]
 815        else:  # Use the solution site
 816            sol_list = [self.sol_int, self.sol_ext]
 817
 818        for sol in sol_list:
 819            if sol is None:  # When there is no exterior domain
 820                continue
 821            if sol_min is not None:
 822                sol[np.where(sol < sol_min)[0]] = sol_min  # Apply lower bound
 823            if sol_max is not None:
 824                sol[np.where(sol > sol_max)[0]] = sol_max  # Apply upper bound

 861    def evaluate_residual_vector(self, vec: Union[None, np.ndarray] = None,
 862                                 force_reassemble=False) -> np.ndarray:
 863        """
 864        Evaluate the residual vector using the user-provided WeakForm instance
 865        `wf_residual`.
 866
 867        Parameters
 868        ----------
 869        vec : Union[None, np.ndarray], optional
 870            If specified, `vec` is the solution vector used to compute the
 871            residual vector. The default is None.
 872        force_reassemble : bool
 873            Force the reassembly of matrices and vectors associated with terms
 874            tagged 'mod'. Is relevant only when `vec` is None.
 875            The default is False.
 876
 877        Returns
 878        -------
 879        residual_vector : np.ndarray
 880            The residual vector.
 881
 882        See Also
 883        --------
 884        evaluate_residual_vector_parts
 885
 886        """
 887
 888        # Reassemble matrices and vectors
 889        if vec is not None:
 890            self.update_nonlinear_mtx_vec('wf_residual', vec=vec)
 891        else:
 892            vec = self.sol_int
 893            if force_reassemble:
 894                self.update_nonlinear_mtx_vec('wf_residual')
 895
 896        # Compute the residual vector using 'no_bc' mtx/vec
 897        wf_res = self.wf_dict['wf_residual']
 898        if wf_res.mtx_vec.mtx_no_bc is None:
 899            wf_res.set_mtx_vec_no_bc()
 900        mtx_res = wf_res.mtx_vec.mtx_no_bc
 901        rhs_res = wf_res.mtx_vec.vec_no_bc
 902        residual_vector = mtx_res.dot(vec) - rhs_res
 903
 904        return residual_vector

 907                                       vec: Union[None, np.ndarray] = None,
 908                                       force_reassemble=False) -> dict:

 911            $$ R = A u_k - rhs_{\text{cst}} - rhs_{\text{mod}} $$
 912
 913        Returns
 914        -------
 915        residual_vector_parts : dict
 916            Dictionary with keys ('mtx_term', 'rhs_cst_term', 'rhs_mod_term')

 919        See Also
 920        --------
 921        evaluate_residual_vector

 924        # Reassemble matrices and vectors
 925        if vec is not None:

 930                self.update_nonlinear_mtx_vec('wf_residual')

 933        wf_res = self.wf_dict['wf_residual']

 935            wf_res.set_mtx_vec_no_bc()

 946        return residual_vector_parts

1021            self, *wf_keys: str, vec: Union[None, np.ndarray] = None):

1074    def _constant_relaxation(self, w=None):
1075        r"""
1076        Use a constant relaxation parameter:
1077        $$ U_{k+1} \gets \omega U_{k+\omega} + (1-\omega) U_k $$
1078
1079        Parameters
1080        ----------
1081        w : Union[None, float], optional
1082            Relaxation parameter to be used. The default is None, is which case
1083            `self.relax_param` is used.
1084        """
1085        w = self.relax_param if w is None else w
1086        self.sol_int = w * self.buffer_int + (1 - w) * self.old_int
1087        if self.sol_ext is not None:
1088            self.sol_ext = w * self.buffer_ext + (1 - w) * self.old_ext

1090    def _line_search_relaxation(self):
1091        """
1092        Use the line-search method to find an optimal relaxation parameter (by
1093        minimizing the 2-norm of the strong residual squared). The current
1094        implementation uses Scipy's `minimize_scalar` bounded method.
1095        """
1096
1097        old_sol = self.old_int
1098        new_sol = self.buffer_int
1099
1100        def func(w: float):
1101            """Scalar function to be minimized"""
1102            vec = w * new_sol + (1 - w) * old_sol
1103            res_vec = self.evaluate_residual_vector(vec)
1104            return np.sum(res_vec**2)
1105
1106        options = {'maxiter': 10, 'xatol': 5e-2}
1107        optim_res = minimize_scalar(func, bounds=(0, 2), method='bounded',
1108                                    options=options)
1109        w = optim_res.x if optim_res.success else None
1110        self._constant_relaxation(w=w)
1111        self._write_linesearch_message(optim_res)
